  DECLARE
  v_exists INTEGER;
BEGIN
  SELECT COUNT(*)
  INTO v_exists
  FROM ALL_PROCEDURES
  WHERE OBJECT_NAME = 'MI_PAQUETE'
    AND PROCEDURE_NAME = 'MI_PROCEDIMIENTO'
    AND OWNER = 'TU_ESQUEMA';

  IF v_exists > 0 THEN
    DBMS_OUTPUT.PUT_LINE('Existe');
  ELSE
    DBMS_OUTPUT.PUT_LINE('No existe');
  END IF;
END;
